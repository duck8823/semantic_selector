# Setup

```bash
brew install mysql pyenv mecab mecab-ipadic

mysql -uxxx -p -e 'create database register_form'
mysql -uxxx -p register_form < data/register_form.sql

# python3.5+
pyenv install 3.5.2
pyenv local 3.5.2
python -m venv venv
source venv/bin/activate
pip3 install -r projects/requirements.txt

# use juman++
brew install boost
curl -O http://lotus.kuee.kyoto-u.ac.jp/nl-resource/jumanpp/jumanpp-1.02.tar.xz
tar jxvf jumanpp-1.02.tar.xz
cd jumanpp
./configure
make
make install

curl -O http://lotus.kuee.kyoto-u.ac.jp/nl-resource/pyknp/pyknp-0.3.tar.gz
pip3 install pyknp-0.3.tar.gz
```
